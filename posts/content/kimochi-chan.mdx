import type { Metadata } from "next";
import '../styles/global.scss';
import '@creative-kit/react/dist/index.css';

import localFont from 'next/font/local'

const RIDIFont = localFont({src: '../public/fonts/RIDIBatang.otf'})

export const metadata: Metadata = {
    title: "Create Next App",
    description: "Generated by create next app",
};

export default function RootLayout({
  children,
}: Readonly<{
    children: React.ReactNode;
}>) {
    return (
        <html lang="ko">
            <body className={RIDIFont.className}>
            <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/github-markdown-css/5.2.0/github-markdown.min.css"></link>
                {children}
            </body>
        </html>
    );
}
